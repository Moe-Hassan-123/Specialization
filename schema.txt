CREATE TABLE students (id INTEGER PRIMARY KEY,
                       name TEXT NOT NULL,
                       password BLOB NOT NULL,
                       national_id BLOB NOT NULL,
                       grade INT NOT NULL,
                       specialization TEXT NOT NULL,
                       CHECK (grade IN (1, 2)),
                       CHECK (specialization IN ("s",
                                                 "l",
                                                 "m",
                                                 "o")));
